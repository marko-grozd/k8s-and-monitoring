# installing loki with helm

helm install loki grafana/loki \
  --namespace monitoring \
  --set loki.auth_enabled=false \
  --set loki.commonConfig.replication_factor=1 \
  --set loki.storage.type=filesystem \
  --set deploymentMode=SingleBinary \
  --set singleBinary.replicas=1 \
  --set read.replicas=0 \
  --set write.replicas=0 \
  --set backend.replicas=0 \
  --set monitoring.selfMonitoring.enabled=false \
  --set monitoring.selfMonitoring.grafanaAgent.installOperator=false \
  --set test.enabled=false \
  --set loki.useTestSchema=true

# Nodes

kubectl get nodes                          # Lista nodova
kubectl get nodes -o wide                  # Sa IP adresama i detaljima
kubectl describe node <ime>               # Detalji noda
kubectl top nodes                          # CPU/RAM po nodu
kubectl label node <ime> role=monitoring  # Dodaj label

#Pods

kubectl get pods -n <namespace>            # Lista Podova
kubectl get pods -n <namespace> -o wide    # Sa nodom i IP-om
kubectl get pods -A                        # Svi Podovi u svim namespacima
kubectl describe pod <ime> -n <namespace> # Detalji Poda
kubectl logs -l app=<ime> -n <namespace>  # Logovi po labelu
kubectl logs <pod> -n <namespace> --previous  # Logovi prethodnog Poda
kubectl top pods -n <namespace>           # CPU/RAM po Podu
kubectl delete pod <ime> -n <namespace>   # Restart Poda

#Deployments
kubectl get deployments -n <namespace>
kubectl rollout restart deployment/<ime> -n <namespace>   # Restart
kubectl rollout status deployment/<ime> -n <namespace>    # Prati status
kubectl scale deployment <ime> --replicas=3 -n <namespace> # Skaliraj

#helm

helm list -n <namespace>                   # Lista instaliranih chartova
helm install <ime> <chart> -f values.yaml  # Instalacija
helm upgrade <ime> <chart> -f values.yaml  # Update
helm upgrade --install <ime> <chart> -f values.yaml  # Install ili update
helm uninstall <ime> -n <namespace>        # Deinstalacija
helm get values <ime> -n <namespace>       # Trenutne vrednosti
helm show values <chart>                   # Default vrednosti charta
helm history <ime> -n <namespace>          # Istorija deployeva
helm rollback <ime> <revision> -n <namespace>  # Rollback
helm upgrade <ime> <chart> --dry-run       # Test bez primene
