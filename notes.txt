minikube start

kubectl port-forward svc/backend-service 8000:8000 -n skeleton &
kubectl port-forward svc/frontend-service 3000:80 -n skeleton &

# installing loki with helm

helm install loki grafana/loki \
  --namespace monitoring \
  --set loki.auth_enabled=false \
  --set loki.commonConfig.replication_factor=1 \
  --set loki.storage.type=filesystem \
  --set deploymentMode=SingleBinary \
  --set singleBinary.replicas=1 \
  --set read.replicas=0 \
  --set write.replicas=0 \
  --set backend.replicas=0 \
  --set monitoring.selfMonitoring.enabled=false \
  --set monitoring.selfMonitoring.grafanaAgent.installOperator=false \
  --set test.enabled=false \
  --set loki.useTestSchema=true
